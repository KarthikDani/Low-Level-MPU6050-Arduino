217bool i2c_write_byte(uint8_t data) {
218 // Loop through each of the 8 bits, MSB first.
219 for (int i = 7; i >= 0; i--) {
220 // Set the SDA line based on the current bit.
221 if ((data >> i) & 0x01) {
222 i2c_sda_high();
223 } else {
224 i2c_sda_low();
225 }
226 delayMicroseconds(I2C_DELAY_US);
227
228 // Clock the bit out by pulsing SCL high.
229 i2c_scl_high();
230 delayMicroseconds(I2C_DELAY_US);
231 i2c_scl_low();
232 delayMicroseconds(I2C_DELAY_US);
233 }
234
235 // The 9th clock pulse is for the ACK/NACK.
236 // The slave will pull SDA low for an ACK.
237 i2c_sda_high(); // Release the SDA line.
238 delayMicroseconds(I2C_DELAY_US);
239 i2c_scl_high();
240 delayMicroseconds(I2C_DELAY_US);
241 bool ack = !i2c_read_sda(); // Read the ACK bit.
242 i2c_scl_low();
243 delayMicroseconds(I2C_DELAY_US);
244 i2c_sda_low(); // Restore SDA to low for the next operation.
245
246 return ack;
247}

261uint8_t i2c_read_byte(bool ack) {
262 uint8_t data = 0;
263 i2c_sda_high(); // Release the SDA line to allow the slave to drive it.
264
265 // Read 8 bits.
266 for (int i = 7; i >= 0; i--) {
267 i2c_scl_high();
268 delayMicroseconds(I2C_DELAY_US);
269 if (i2c_read_sda()) {
270 data |= (1 << i);
271 }
272 i2c_scl_low();
273 delayMicroseconds(I2C_DELAY_US);
274 }
275
276 // Send ACK or NACK.
277 if (ack) {
278 i2c_sda_low(); // Pull SDA low for an ACK.
279 } else {
280 i2c_sda_high(); // Keep SDA high for a NACK.
281 }
282 delayMicroseconds(I2C_DELAY_US);
283 i2c_scl_high();
284 delayMicroseconds(I2C_DELAY_US);
285 i2c_scl_low();
286 delayMicroseconds(I2C_DELAY_US);
287 i2c_sda_low(); // Restore SDA low for the next operation.
288
289 return data;
290}

344void mpu6050_read_bytes(uint8_t reg, uint8_t* buffer, uint8_t count) {
345 i2c_start();

347 i2c_write_byte(reg); // Starting register address
348 i2c_stop();
349
350 i2c_start();
351 i2c_write_byte(MPU6050_ADDRESS << 1 | 0x01); // Read address (1)
352 for (int i = 0; i < count; i++) {
353 buffer[i] = i2c_read_byte(i < count - 1); // ACK for all but the last byte
354 }
355 i2c_stop();

396void mpu6050_read_accel_g(accel_float_data_t* accel_g) {
397 accel_data_t raw_accel;
398 mpu6050_read_accel_data(&raw_accel);
399
400 // The MPU6050 is configured for +/-2g range, with a resolution of 16 bits.
401 // The conversion factor is 2.0g / 32768.0 LSB.
402 const float scale_factor = 2.0 / 32768.0;
403
404 accel_g->x = (float)raw_accel.x * scale_factor;
405 accel_g->y = (float)raw_accel.y * scale_factor;
406 accel_g->z = (float)raw_accel.z * scale_factor;
407}

438void mpu6050_read_temp_celsius(temp_float_data_t* temp_c) {
439 uint8_t buffer[2];
440 mpu6050_read_bytes(MPU6050_TEMP_OUT_H_REG, buffer, 2);
441
442 int16_t raw_temp = (int16_t)(buffer[0] << 8 | buffer[1]);
443
444 // The conversion formula for MPU6050 temperature is from the datasheet:
445 // Temperature in C = ((raw_temp) / 340.0) + 36.53
446 temp_c->c = ((float)raw_temp / 340.0) + 36.53;
447}

464void setup() {
465 Serial.begin(115200);
466 while (!Serial); // Wait for the Serial port to connect.
467
468 // Set the SCL and SDA pins as outputs initially, and then let the
469 // I2C functions manage them.
470 pinMode(I2C_SDA_PIN, OUTPUT);
471 pinMode(I2C_SCL_PIN, OUTPUT);
472
473 // Initialize the MPU6050.
474 // The sensor starts in sleep mode, so we need to wake it up.
475 // We'll write a 0 to the PWR_MGMT_1 register to wake it up.
476 mpu6050_write_register(MPU6050_PWR_MGMT_1_REG, 0);
477
478 // Verify the MPU6050 is connected by reading the WHO_AM_I register.
479 // It should return 0x68.
480 uint8_t who_am_i = mpu6050_read_register(MPU6050_WHO_AM_I_REG);
481 if (who_am_i != MPU6050_ADDRESS) {
482 Serial.print("MPU6050 not found! WHO_AM_I returned: 0x");
483 Serial.println(who_am_i, HEX);
484 while (1); // Halt the program if the sensor is not found.
485 } else {
486 Serial.println("MPU6050 found and initialized successfully.");
487 }
488}
